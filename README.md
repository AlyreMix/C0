     * IF_KW     -> 'if'

     * MUL       -> '*'
     * DIV       -> '/'
     * ASSIGN    -> '='
     * EQ        -> '=='
     * NEQ       -> '!='
     * LT        -> '<'
     * GT        -> '>'
     * LE        -> '<='
     * GE        -> '>='
     * L_PAREN   -> '('
     * R_PAREN   -> ')'
     * L_BRACE   -> '{'
     * R_BRACE   -> '}'
     * ARROW     -> '->'
     * COMMA     -> ','
     * COLON     -> ':'
     * SEMICOLON -> ';'
     * IDENT     -> 'ident'
     * UINT_LITERAL -> 'uint_literal'
